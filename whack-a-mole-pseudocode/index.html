define a state as:
  a 4x4 grid containing none or more moles,
  a list describing the steps to get to this state.

define enqueue(E,Q) as:
  adding the element E to the queue Q.

define dequeue(Q) as:
  obtaining the element from the front of the queue Q.

create empty queue Q

enqueue(initial state, Q)

while Q is not empty:

  S <- dequeue(Q)

  if the grid stored in S is empty:
    print the steps to get to S from initial state
  else:

    for each mole M in the grid stored in S:
      create a new state S' where M has been hit
      add the mole to the steps of S'
      enqueue S'
